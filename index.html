Quick Start

# Clone and setup git clone https://github.com/abhiiibabariya-dev/threat-intelligence-blueteam.git cd threat-intelligence-blueteam pip install -r requirements.txt # Collect IOCs from 7 threat feeds python ioc/ioc_manager.py collect # Analyze suspicious email headers python scripts/email_analyzer.py -f suspicious_headers.txt # Check IP reputation python scripts/ip_reputation.py 8.8.8.8 1.2.3.4 # Search CVE vulnerabilities + CISA KEV python scripts/cve_checker.py search "apache log4j" python scripts/cve_checker.py kev --recent 20 # Generate forensic timeline python scripts/timeline_generator.py /var/log/auth.log /var/log/syslog # Map MITRE ATT&CK coverage and find gaps python scripts/attack_mapper.py --gaps-only # Generate threat intelligence report python scripts/threat_report.py --markdown report.md # Analyze logs for threats python scripts/log_analyzer.py --input /var/log/syslog # Network baseline and scan python scripts/network_scan.py baseline python scripts/network_scan.py scan
