# Securonix Spotter Queries - Credential Access
# MITRE ATT&CK Tactic: TA0006
# Usage: Run in Securonix Spotter search interface

# SQ-CA-001: Brute Force Detection (T1110.001)
index = activity AND riskThreatName = "Brute Force" AND accountname != "SYSTEM"
| where eventoutcome = "Failure"
| stats count() as failed_attempts, dc(accountname) as unique_accounts by sourceaddress
| where failed_attempts > 10
| sort -failed_attempts
| fields sourceaddress, failed_attempts, unique_accounts, datetime

# SQ-CA-002: Password Spray Attack (T1110.003)
index = activity AND eventoutcome = "Failure" AND categoryobject = "Authentication"
| stats dc(accountname) as target_accounts, count() as attempts by sourceaddress
| where target_accounts > 5 AND attempts > 10
| sort -target_accounts
| fields sourceaddress, target_accounts, attempts

# SQ-CA-003: LSASS Memory Dump (T1003.001)
index = activity AND objectname CONTAINS "lsass" AND categoryobject = "Process Access"
| where resourcecustomfield1 IN ("0x1010", "0x1038", "0x1fffff")
| where sourceprocessname NOT IN ("csrss.exe", "svchost.exe", "MsMpEng.exe")
| fields datetime, devicename, sourceprocessname, objectname, resourcecustomfield1

# SQ-CA-004: Kerberoasting Detection (T1558.003)
index = activity AND baseeventid = "4769" AND resourcecustomfield5 IN ("0x17", "0x18")
| where accountname NOT ENDS WITH "$"
| stats count() as tgs_requests by accountname, sourceaddress
| where tgs_requests > 3
| fields accountname, sourceaddress, tgs_requests, datetime

# SQ-CA-005: DCSync Attack (T1003.006)
index = activity AND baseeventid = "4662"
| where resourcecustomfield1 CONTAINS "1131f6aa-9c07-11d1-f79f-00c04fc2dcd2" OR resourcecustomfield1 CONTAINS "1131f6ad-9c07-11d1-f79f-00c04fc2dcd2"
| where sourceaddress NOT IN ($domain_controllers)
| fields datetime, accountname, sourceaddress, devicename

# SQ-CA-006: Credential File Access (T1552.001)
index = activity AND categoryobject = "File Access"
| where objectname MATCHES ".*(unattend\.xml|sysprep\.inf|\.aws.credentials|web\.config|shadow|\.git-credentials).*"
| fields datetime, accountname, devicename, objectname, sourceprocessname

# SQ-CA-007: SAM Registry Dump (T1003.002)
index = activity AND categoryobject = "Process" AND categoryoutcome = "Created"
| where resourcecustomfield2 CONTAINS "reg" AND resourcecustomfield2 CONTAINS "save" AND (resourcecustomfield2 CONTAINS "sam" OR resourcecustomfield2 CONTAINS "security" OR resourcecustomfield2 CONTAINS "system")
| fields datetime, accountname, devicename, resourcecustomfield2

# SQ-CA-008: Mimikatz Execution (T1003)
index = activity AND categoryobject = "Process"
| where resourcecustomfield2 MATCHES ".*(mimikatz|sekurlsa|kerberos::list|privilege::debug|lsadump).*"
| fields datetime, accountname, devicename, sourceprocessname, resourcecustomfield2
